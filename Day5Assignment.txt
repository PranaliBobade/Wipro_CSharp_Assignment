using System;
namespace AssignmentDayFive{
class Program
{
    static void Main()
    {

        int number = 5;
        Console.WriteLine($"Factorial of {number} is: {Factorial(number)}");

      
        string inputString = "Welcome to C# Programming";
        Console.WriteLine("Printing each word in the string:");
        PrintEachWord(inputString);

       
        int numberToSum = 12345;
        Console.WriteLine($"Sum of alternate digits of {numberToSum} is: {SumAlternateDigits(numberToSum)}");

        int a = 24, b = 36;
        Console.WriteLine($"GCD of {a} and {b} is: {GCD(a, b)}");
    }

  
    static int Factorial(int n)
    {
        if (n == 0)
            return 1;
        int result = 1;
        for (int i = 1; i <= n; i++)
        {
            result *= i;
        }
        return result;
    }

    static void PrintEachWord(string input)
    {
        string[] words = input.Split(' ');
        foreach (string word in words)
        {
            Console.WriteLine(word);
        }
    }

   
    static int SumAlternateDigits(int number)
    {
        int sum = 0;
        int count = 1; 
        while (number > 0)
        {
            int digit = number % 10;
            if (count % 2 != 0)
            {
                sum += digit;
            }
            number /= 10;
            count++;
        }
        return sum;
    }

    static int GCD(int a, int b)
    {
        while (b != 0)
        {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
 }
}
